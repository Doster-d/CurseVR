86 private void HandleVoiceInactive(AudioClip clip)
87 {
88 if (!isProcessingVoice || clip == null) return;
89
90 // Convert AudioClip to byte array
91 float[] samples = new float[clip.samples * clip.channels];
92 clip.GetData(samples, 0);
93
94 // Convert to 16-bit PCM
95 byte[] audioData = new byte[samples.Length * 2];
96 for (int i = 0; i < samples.Length; i++)
97 {
98 short value = (short)(samples[i] * 32767f);
99 audioData[i * 2] = (byte)(value & 0xff);
100 audioData[i * 2 + 1] = (byte)((value >> 8) & 0xff);
101 }
102
103 // Send to voice service
104 _ = voiceService.SendAudioDataAsync(audioData);
105
106 // Play debug audio if needed
107 if (debugAudioSource != null)
108 {
109 debugAudioSource.clip = clip;
110 debugAudioSource.Play();
111 }
112 }

114 private void HandleVoiceCommand(VoiceCommandData commandData)
115 {
116 Debug.Log($"Voice command recognized: {commandData.CommandType}");
117
118 if (actionReferences.TryGetValue(commandData.CommandType, out var actionRef))
119 {
120 var action = actionRef.action;
121 if (action != null)
122 {
123 Debug.Log($"Triggering input action: {action.name}");
124
125 // Simply enable the action - this will trigger any subscribers
126 // that are monitoring for this action's state changes
127 action.Enable();
128
129 // This is a workaround - Unity's InputSystem doesn't provide
130 // a public API to directly trigger an action from code
131 // We're just toggling the action's state via enable/disable
132 StartCoroutine(ToggleAction(action));
133 }
134 }
135 else
136 {
137 Debug.LogWarning($"No action found for command type: {commandData.CommandType}");
138 }
139 }
